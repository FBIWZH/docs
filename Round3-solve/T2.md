## $T2 \ 求和$

非常模板。

树状数组即可~~(你自己作死用线段树不关我事)~~

```cpp
#include <cstdio>
#include <iostream>
using namespace std;
#define int long long
int t[1000005],n;
inline int lowbit(int x){return x&-x;}
inline void add(int index,int k)
{
    while(index<=n)
        t[index]+=k,index+=lowbit(index);
}
inline int getsum(int x)
{
    int ans=0;
    while(x>0)
        ans+=t[x],x-=lowbit(x);
    return ans;
}
inline int getsum(int l,int r){return getsum(r)-getsum(l-1);}
signed main()
{
    int m;
    scanf("%lld%lld",&n,&m);
    for(int i=1;i<=n;i++)
    {
        int x;
        scanf("%lld",&x);
        add(i,x);
    }
    for(int i=1;i<=m;i++)
    {
        string op;
        cin>>op;
        if(op=="SET")
        {
            int x,k;
            scanf("%lld,%lld",&x,&k);
            int t=getsum(x,x);
            add(x,k-t);
        }
        else
        {
            int l,r;
            scanf("%lld,%lld",&l,&r);
            if(l>r)swap(l,r);
            printf("%lld\n",getsum(l,r));
        }
    }
    return 0;
}
```
